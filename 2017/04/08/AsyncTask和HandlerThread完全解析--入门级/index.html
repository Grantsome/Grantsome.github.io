жң¬жқҘжү“з®—жҜҸе‘ЁеҶҷдёҖзҜҮеҚҡе®ўзҡ„пјҢдҪҶжҳҜдёҠдёҖе‘ЁжІЎжңүеҶҷпјҢеҺҹеӣ еҫҲз®ҖеҚ•пјҡж”ҫжё…жҳҺдёүеӨ©еҺ»зҺ©дәҶпјҢзј“дәҶеҘҪеҮ еӨ©еӣһиҝҮзҘһжқҘгҖӮжүҖд»ҘиҝҷдёҖе‘Ёжү“з®—еҶҷдёӨзҜҮпјҢе°ұеҪ“еҒҡжҳҜиҮӘе·ұеҜ№иҮӘе·ұзҡ„дёҖдёӘдәӨд»Јеҗ§гҖӮиҜқдёҚеӨҡиҜҙпјҢиҝӣе…ҘжӯЈйўҳгҖӮ

AsyncTaskпјҢзҝ»иҜ‘иҝҮжқҘзҡ„еӯ—йқўж„ҸжҖқжҳҜејӮжӯҘд»»еҠЎ,жҲ‘д»¬д№ жғҜдёҠиҜҙејӮжӯҘж“ҚдҪң(жҚўеҸҘиҜқиҜҙпјҢд№ҹе°ұжҳҜеңЁдё»зәҝзЁӢдёӯж–°ејҖзәҝзЁӢиҝӣиЎҢзҡ„ж“ҚдҪң)гҖӮз®ҖеҚ•жҜ”е–»пјҡеҒҮи®ҫдҪ йҘҝдәҶпјҢдҪ иӮҜе®ҡдјҡеҺ»йҘӯеә—ең°ж–№еҗғдёңиҘҝпјҢдҪҶжҳҜйҘӯеә—дёәдҪ еҮҶеӨҮйЈҹзү©йңҖиҰҒдёҖе®ҡзҡ„ж—¶й—ҙпјҢжүҖд»ҘдҪ йҘҝдәҶеҝ…йЎ»иҰҒеҲ°дәҶйҘӯеә—зӯүдёҖдјҡе„ҝжүҚиғҪеҗғйҘӯпјҢиҝҷе°ұжҳҜејӮжӯҘж“ҚдҪңпјҢдҪ дёҚиғҪй©¬дёҠеҒҡдёӢдёҖд»¶дәӢжғ…пјҢеӣ дёәе®ғйңҖиҰҒеҪ“еүҚдәӢжғ…е®ҢжҲҗд№ӢеҗҺжүҚиғҪеҒҡдёӢдёҖд»¶дәӢжғ…гҖӮиҮідәҺдёәд»Җд№ҲйңҖиҰҒејӮжӯҘж“ҚдҪңпјҢAndroidдё»зәҝзЁӢйҮҢйқўдёҚиғҪиҝӣиЎҢиҖ—ж—¶ж“ҚдҪңпјҢд№ҹе°ұжҳҜеҸӘиғҪеңЁеӯҗзәҝзЁӢйҮҢиҝӣиЎҢиҖ—ж—¶ж“ҚдҪңжҜ”еҰӮзҪ‘з»ңиҜ·жұӮпјҢе№¶дё”еҸӘиғҪеңЁдё»зәҝзЁӢйҮҢйқўжӣҙж–°UIгҖӮз®ҖеҚ•дҫӢеӯҗпјҢд»ҺзҪ‘дёҠеҠ иҪҪеӣҫзүҮж•°жҚ®е№¶жҳҫзӨәеӣҫзүҮпјҢзҺ°еңЁдҪ иҰҒйҖҡиҝҮзҪ‘з»ңиҜ·жұӮиҺ·еҸ–дёҖж®өж•°жҚ®з»ҷдҪ пјҢе…¶дёӯжңүеӣҫзүҮзҡ„urlпјҢеҫ—еҲ°urlиҪ¬еҢ–жҲҗbitmapжҲ–иҖ…drawableпјҢз„¶еҗҺеңЁеӣһеҲ°дё»зәҝзЁӢйҮҢйқўз»ҷimageviewи®ҫзҪ®еӣҫзүҮгҖӮ

иҜҙе®ҢдәҶAysncTaskжҲ‘д»¬еҶҚжқҘи®Ёи®әдёҖдёӢе®ғзҡ„дјҳзјәзӮ№пјҢиҝҷдёӘжҠҪиұЎзҡ„е·Ҙе…·зұ»иҷҪз„¶еҘҪз”ЁпјҢдҪҶжҳҜе®ғд№ҹеҸӘжҳҜдёҖз§ҚиҪ»йҮҸзә§зҡ„е·Ҙе…·иҖҢе·ІпјҢеҸӘйҖӮеҗҲеҸӘжңүеҮ з§’зҡ„зҹӯж—¶й—ҙзҡ„ејӮжӯҘж“ҚдҪңиҖҢе·ІпјҢе№¶дё”жңүеҸҜиғҪеҜјиҮҙеҶ…еӯҳжі„жјҸзҡ„й—®йўҳгҖӮжҲ‘д»¬еҪ“з„¶иҰҒиҰҒеҺ»еҜ»жүҫе…¶д»–зҡ„жӣҝд»Јзҡ„дёңиҘҝд»ҘеҶҷеҮәжӣҙеҘҪзҡ„зЁӢеәҸгҖӮ

иҝҷж—¶еҖҷLooperгҖҒHandlerгҖҒHandlerThreadе°ұй—Әдә®зҷ»еңәдәҶгҖӮ
Message:жұүиҜӯзҝ»иҜ‘иҝҮжқҘжҳҜж¶ҲжҒҜзҡ„ж„ҸжҖқпјҢйЎҫеҗҚжҖқд№үпјҢе°ұжҳҜеңЁзәҝзЁӢд№Ӣй—ҙдј йҖ’зҡ„ж¶ҲжҒҜпјҢеҸҜд»ҘжҗәеёҰmessage.whatзҡ„intзұ»еһӢ,д№ҹеҸҜд»ҘжҗәеёҰдёҖдёӘobjectзұ»еһӢзҡ„obj.
Handler:еӨ„зҗҶиҖ…,дё»иҰҒз”ЁдәҺеҸ‘йҖҒе’ҢеӨ„зҗҶж¶ҲжҒҜпјҢеҸ‘йҖҒжҳҜhandler.sendMessage()пјҢжңҖз»Ҳдј йҖ’еҲ°handler.handlerMessage();иҜҙз®ҖеҚ•дёҖзӮ№е°ұжҳҜејӮжӯҘжӣҙж–°UIпјҢе№¶дё”жӣҙж–°UIзҡ„ж—¶еҖҷпјҢжҳҜдё»зәҝзЁӢдёҺеӯҗзәҝзЁӢд№Ӣй—ҙзҡ„йҖҡдҝЎгҖӮ

    //е®һдҫӢеҢ–ConcurrentHashMap    private ConcurrentMap<T,String> mRequestMap = new ConcurrentHashMap<>();    private boolean mHashQuit = false;
    //еЈ°жҳҺиҜҘжҺҘеҸЈ    private ThumbnailDownloaderListener<T> mThumbnailDownloader;
    //иҮӘе®ҡд№үдёҖдёӘжҺҘеҸЈ    public interface ThumbnailDownloaderListener<T>{        void onThumbnailDownloaded(T target,Bitmap thumbnail);    }

    //е®һзҺ°дёҖдёӘhandler    @Override    protected void onLooperPrepared(){        mRequestHandler = new Handler(){           @Override            public void handleMessage(Message msg){               if(msg.what == MESSAGE_DOWNLOAD){                   T target = (T) msg.obj;                   Log.i(TAG,"Got a request for URL: " + mRequestMap.get(target));                   handleRequest(target);               }           }        };    }    private void handleRequest(final T target){        try{            final String url = mRequestMap.get(target);            if(url==null){                return;            }            byte[] bitmapBytes = new HttpUtil().getUrlBytes(url);            final BitmapFactory.Options options = new BitmapFactory.Options();            options.inSampleSize = 3;            final Bitmap bitmap = BitmapFactory.decodeByteArray(bitmapBytes,0,bitmapBytes.length,options);            Log.i(TAG,"Bitmap created");            //дё»зәҝзЁӢ            mRequestHandler.post(new Runnable() {                @Override                public void run() {                    if(mRequestMap.get(target)!=url||mHashQuit){                        return;                    }                    mRequestMap.remove(target);                    mThumbnailDownloader.onThumbnailDownloaded(target,bitmap);                }            });        }catch (Exception e){            e.printStackTrace();        }    }    @Override    public boolean quit(){        mHashQuit = true;        return super.quit();    }    public void queueThumbnail(T target,String url){        Log.i(TAG,"Got a URL " + url);        if(url == null){            mRequestMap.remove(target,url);        }else {            mRequestMap.put(target,url);            mRequestHandler.obtainMessage(MESSAGE_DOWNLOAD,target).sendToTarget();        }    }    public void clearQueue(){        mRequestHandler.removeMessages(MESSAGE_DOWNLOAD);    }
